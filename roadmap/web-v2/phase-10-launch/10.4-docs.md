### Interactive Examples

```tsx
// src/components/docs/InteractiveExample.tsx
'use client';

import { useState } from 'react';
import { Card } from '@/components/ui/card';
import { Button } from '@/components/ui/button';

export function WorkflowExample() {
  const [step, setStep] = useState(0);

  const steps = [
    { title: 'Identify Problem', description: 'What problem are you solving?' },
    { title: 'Define Solution', description: 'How will you solve it?' },
    { title: 'Validate Assumptions', description: 'Test your hypothesis' }
  ];

  return (
    <Card className="p-6">
      <h3 className="font-semibold mb-4">Try a Workflow</h3>

      <div className="space-y-4">
        <div>
          <h4 className="font-medium">{steps[step].title}</h4>
          <p className="text-sm text-muted-foreground">{steps[step].description}</p>
        </div>

        <div className="flex gap-2">
          <Button
            variant="outline"
            onClick={() => setStep(Math.max(0, step - 1))}
            disabled={step === 0}
          >
            Previous
          </Button>

          <Button
            onClick={() => setStep(Math.min(steps.length - 1, step + 1))}
            disabled={step === steps.length - 1}
          >
            Next
          </Button>
        </div>
      </div>
    </Card>
  );
}
```
